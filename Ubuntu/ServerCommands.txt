sudo docker run -d \
  --hostname ubuntu \
  --name my-ubuntu-server \
  -p 2222:22 \

  -v ubuntu_etc:/etc \
  -v ubuntu_home:/home \
  -v ubuntu_lib:/lib \
  -v ubuntu_usr:/usr \
